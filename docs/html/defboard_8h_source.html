259  std::tuple<bool, Key, std::shared_ptr<T>, std::list<Position>>
260  getElement(const Position &position) {
261  auto position_w = aux_ToInnerPosition(PositionTypeNormal, position);
262  auto positiondef_w = aux_ToInnerPosition(PositionTypeDefault, position);
263  auto defaultid_w = std::make_tuple(this->defaultid, ElementTypeDefault);
264  bool isNormal = true;
265 
266  if (!this->Base::aux_checkLimits(position_w))
267  throw PositionOutLimits<Key, Position>(this->name, position);
268 
269  /* Looks for a normal element */
270  auto info = this->Base::aux_checkOccupied(position_w);
271  if (!std::get<0>(info)) {
272  /* If there is no normal element, looks for custom default element */
273  isNormal = false;
274  info = this->Base::aux_checkOccupied(positiondef_w);
275  if (!std::get<0>(info)) {
276  /* If there is neither a custom default element, uses the master default
277  * element */
278  info = this->Base::aux_checkIDExists(defaultid_w);
279  }
280  }

282  std::list<Position> positionlist;
283  for (typename Base::Position i : std::get<3>(info)) {
284  positionlist.push_back(aux_ToPublicPosition(i));
285  }
286

288  isNormal,
289  std::get<0>(std::get<1>(info) /* Inner get() returns (id,elemType) */),
